## 例子

```js
// 20题
// 左右括号的合法性 [({})]
function isValid(s) {
  var stack = [];
  var parenMap = {
    ')': '(',
    ']': '[',
    '}': '{'
  };
  for (var i = 0; i < s.length; i++) {
    var c = s[i];
    if (!parenMap[c]) {
      stack.push(c);
    } else if (parenMap[c] !== stack.pop()) {
      return false;
    }
  }
  return true;
}
```
